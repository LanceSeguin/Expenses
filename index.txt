// --- CONFIG & SCOUT LIST ---
const HARDCODED_WEB_APP_URL = 'https://script.google.com/macros/s/AKfycbzunFqBMn82NpK9Y3R8hO9LBJb3SHLh4VgXKMAe4DNqAvkXyBRBSlZzh-uy7CcH---0ew/exec';
const HARDCODED_VISION_KEY  = 'AIzaSyANA15O0FPREaFOIN5xXszhXlXHapP9Q0Y';

// --- KEYWORD SCOUT LIST ---
// The scanner will prioritize finding these specific strings in your receipt text.
const VENDOR_KEYWORDS = [
  "Home Depot", "Lowe's", "Walmart", "Amazon", "Shell", "Exxon", "Costco", 
  "Sherwin-Williams", "Ferguson", "Target", "Office Depot", "Staples", "Google Pay", 
  "Shell", "Circle K", "Wright", "Racetrac", "Door Dash", "Supra", "National", "Harbor Freight", 
  "Holmes", "Whollesale Flooring", "Chevron", "Window Medics", "Shellpoint"
];

const DEFAULTS = {
  LNI: { cards: ['6247','1334','1326','6254'], props: ['Beau Darren','Briarhurst','Cornerstone','Letitia'] },
  LNEP: { cards: ['3323','4134','4126','3331'], props: ['Rue De Belle Maison','Parkridge'] }
};

let CFG = { webAppUrl: HARDCODED_WEB_APP_URL, visionApiKey: HARDCODED_VISION_KEY, ...DEFAULTS };
const STATE = { fileBase64: null, fileMimeType: null, scanned: {}, business: null, vendor: '', date: null };

// --- SCANNER WITH SCOUT LOGIC ---
async function scanReceiptWithVision() {
  const status = document.getElementById('scanStatus');
  status.style.display = 'flex';
  try {
    const url = `https://vision.googleapis.com/v1/images:annotate?key=${CFG.visionApiKey}`;
    const body = { requests: [{ image: { content: STATE.fileBase64 }, features: [{ type: 'DOCUMENT_TEXT_DETECTION' }] }] };
    const resp = await fetch(url, { method: 'POST', body: JSON.stringify(body) });
    const data = await resp.json();
    const fullText = data.responses[0]?.fullTextAnnotation?.text || "";
    
    // START SCOUTING
    let foundVendor = "";
    for (let kw of VENDOR_KEYWORDS) {
      if (fullText.toLowerCase().includes(kw.toLowerCase())) {
        foundVendor = kw; // Match found!
        break;
      }
    }
    // Fallback: If no keyword matches, use the original line-guessing logic
    if (!foundVendor) {
      const lines = fullText.split('\n').map(l => l.trim()).filter(Boolean);
      for (let line of lines.slice(0, 5)) {
        if (line.length > 3 && !/\d/.test(line)) { foundVendor = line; break; }
      }
    }
    STATE.scanned.vendor = foundVendor;
    
    // Parse Date & Amount (omitted for brevity, keeping original logic)
    const dateMatch = fullText.match(/(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{2,4})/);
    if (dateMatch) STATE.scanned.date = new Date(dateMatch[0]);

  } catch (e) { console.error("Scan failed", e); }
  status.style.display = 'none';
  document.getElementById('uploadNextBtn').disabled = false;
}

// --- TOTALS FETCHING (H1 Logic) ---
async function loadBusinessTotals() {
  const year = new Date().getFullYear();
  try {
    const resp = await fetch(`${CFG.webAppUrl}?action=getBusinessTotals&year=${year}`);
    const data = await resp.json();
    Object.keys(data.totals).forEach(biz => {
      const el = document.getElementById('biz-total-' + biz);
      const val = data.totals[biz];
      el.textContent = (val >= 0 ? '+' : '') + '$' + Math.abs(val).toFixed(2);
      el.className = 'prop-total-badge ' + (val > 0 ? 'prop-total-positive' : val < 0 ? 'prop-total-negative' : 'prop-total-zero');
      el.style.display = 'block';
    });
  } catch (e) { console.log("H1 Totals load failed"); }
}

// UI Helpers (Original logic remains same)
function goToScreen(name) {
  if (name === 'business') loadBusinessTotals();
  if (name === 'vendor') prefillVendor();
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById('screen-' + name).classList.add('active');
}
function prefillVendor() { 
  const input = document.getElementById('vendorInput');
  if (STATE.scanned.vendor) {
    input.value = STATE.scanned.vendor;
    document.getElementById('vendorHint').textContent = "Scout found matching vendor";
    document.getElementById('vendorHint').className = "field-hint scanned";
  }
  checkVendorNext();
}
function checkVendorNext() { document.getElementById('vendorNextBtn').disabled = !document.getElementById('vendorInput').value; }
window.onload = () => goToScreen('upload');
